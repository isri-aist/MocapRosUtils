13 Eigen::Matrix3d rotationMatrixFromZAxis(const Eigen::Vector3d & z_axis)
14 {
15  Eigen::Matrix3d mat;
16  mat.row(2) = z_axis.normalized();
17  mat.row(0) = mat.row(2).cross(Eigen::Vector3d::UnitX());
18  if(mat.row(0).norm() < 1e-3)
19  {
20  mat.row(0) = mat.row(2).cross(Eigen::Vector3d::UnitY());
21  }
22  mat.row(0).normalize();
23  mat.row(1) = mat.row(2).cross(mat.row(0)).normalized();
24  return mat;
25 }

65 bool BvhConverter::parse(const std::string & filename)
66 {
67  if(config_.print_level >= 1)
68  {

70  }
71 
72  // Open file
73  std::ifstream ifs(filename);
74  if(!ifs)
75  {
76  ROS_ERROR_STREAM("[BvhConverter::parse] Cannot open the file: " << filename);
77  return false;
78  }
79 
80  std::string line_str;
81  std::string content_phase = ""; // "HIERARCHY" or "MOTION"
82  std::vector<std::string> hierarchy_str_list;
83  int frame_num = 0;
84  int frame_idx = 0;
85  joint_traj_data_.pos_list_.resize(0, 0);
86  while(std::getline(ifs, line_str, '\n'))
87  {
88  // Treat newline code for Windows (i.e., CRLF)
89  if(line_str.back() == '\r')
90  {
91  line_str.pop_back();
92  }
93 
94  // Skip empty line
95  if(line_str.empty())
96  {
97  continue;
98  }
99 
100  // Replace tab with space
101  std::replace(std::begin(line_str), std::end(line_str), '\t', ' ');
102 
103  // Update content_phase
104  if(line_str == "HIERARCHY")
105  {
106  if(content_phase != "")
107  {
108  ROS_ERROR_STREAM("[BvhConverter::parse] HIERARCHY must be first, but current content_phase is {}"
109  << content_phase);
110  return false;
111  }
112  content_phase = "HIERARCHY";
113  continue;
114  }
115  else if(line_str == "MOTION")
116  {
117  if(content_phase != "HIERARCHY")
118  {
119  ROS_ERROR_STREAM("[BvhConverter::parse] MOTION must come after HIERARCHY, but current content_phase is "
120  << content_phase);
121  return false;
122  }
123  content_phase = "MOTION";
124  continue;
125  }
126  if(content_phase == "")
127  {
128  ROS_ERROR_STREAM("[BvhConverter::parse] content_phase is empty.");
129  return false;
130  }
131 
132  // Store each content
133  if(content_phase == "HIERARCHY")
134  {
135  splitStr(line_str, ' ', std::back_inserter(hierarchy_str_list), true);
136  }
137  else // if(content_phase == "MOTION")
138  {
139  std::vector<std::string> motion_str_list = splitStr(line_str, ' ', true);
140  if(motion_str_list.size() > 1 && motion_str_list[0] == "Frames:")
141  {
142  frame_num = std::stoi(motion_str_list[1]);
143  }
144  else if(motion_str_list.size() > 2 && motion_str_list[0] == "Frame" && motion_str_list[1] == "Time:")
145  {
146  joint_traj_data_.dt_ = std::stod(motion_str_list[2]);
147  }
148  else
149  {
150  if(joint_traj_data_.pos_list_.size() == 0)
151  {
152  joint_traj_data_.pos_list_.resize(frame_num, motion_str_list.size());
153  }
154  else
155  {
156  if(joint_traj_data_.pos_list_.cols() != static_cast<int>(motion_str_list.size()))
157  {
158  ROS_ERROR_STREAM("[BvhConverter::parse] Motion dimensions change: " << joint_traj_data_.pos_list_.cols()
159  << " != " << motion_str_list.size());
160  }
161  }
162  for(int col_idx = 0; col_idx < joint_traj_data_.pos_list_.cols(); col_idx++)
163  {
164  joint_traj_data_.pos_list_(frame_idx, col_idx) = std::stod(motion_str_list[col_idx]);
165  }
166  frame_idx++;
167  }
168  }
169  }
170 
171  // Parse hierarchy
172  parseBvhHierarchy(hierarchy_str_list);
173  if(joint_traj_data_.pos_list_.cols() != static_cast<int>(joint_traj_data_.names_.size()))
174  {
175  ROS_ERROR_STREAM("[BvhConverter::parse] Hierarchy and motion dimensions are inconsistent: "
176  << joint_traj_data_.pos_list_.cols() << " != " << joint_traj_data_.names_.size());
177  }
178 
179  // Convert motion unit
180  for(int col_idx = 0; col_idx < joint_traj_data_.pos_list_.cols(); col_idx++)
181  {
182  if(joint_traj_data_.names_[col_idx].find("rotation") != std::string::npos)
183  {
184  joint_traj_data_.pos_list_.col(col_idx) *= M_PI / 180.0;
185  }
186  else if(joint_traj_data_.names_[col_idx].find("position") != std::string::npos)
187  {
188  joint_traj_data_.pos_list_.col(col_idx) *= config_.pos_scale;
189  }
190  }
191 
192  return true;
193 }

295  }
296 
297  it++;
298  return bvh_joint;

308 {

311 
312  // Add virtual root link

337  rbd_parser_result.mb = mb;

347 void BvhConverter::addRbdBody(const std::shared_ptr<BvhJoint> & bvh_joint,
348  const std::string & parent_name,
349  rbd::MultiBodyGraph & mbg) const
350 {
351  std::string parent_body_name = parent_name;
352  for(size_t i = 0; i < bvh_joint->rbd_joint_list.size(); i++)
353  {
354  // Add joint
355  const auto & rbd_joint = bvh_joint->rbd_joint_list[i];
356  mbg.addJoint(rbd_joint);
357 
358  // Add body
359  bool is_last = (i == bvh_joint->rbd_joint_list.size() - 1);
360  std::string body_name = (is_last ? bvh_joint->name : rbd_joint.name());
361  rbd::Body rbd_body = (is_last ? rbd::Body(1.0, Eigen::Vector3d::Zero(), Eigen::Matrix3d::Identity(), body_name)
362  : rbd::Body(0.0, Eigen::Vector3d::Zero(), Eigen::Matrix3d::Zero(), body_name));
363  mbg.addBody(rbd_body);
364 
365  // Connect bodies with joint
366  sva::PTransformd rel_trans = (i == 0 ? sva::PTransformd(bvh_joint->offset) : sva::PTransformd::Identity());
367  mbg.linkBodies(parent_body_name, rel_trans, body_name, sva::PTransformd::Identity(), rbd_joint.name());
368  parent_body_name = body_name;
369  }
370 
371  for(const auto & child_bvh_joint : bvh_joint->child_joints)
372  {
373  addRbdBody(child_bvh_joint, bvh_joint->name, mbg);
374  }

377 void BvhConverter::setRbdVisual(rbd::parsers::ParserResult & rbd_parser_result) const
378 {
379  for(const auto & bvh_joint : bvh_joint_list_)
380  {
381  std::vector<rbd::parsers::Visual> rbd_visual_list;
382  for(const auto & child_bvh_joint : bvh_joint->child_joints)
383  {
384  if(config_.disable_end_link_visual && child_bvh_joint->is_end)
385  {
386  continue;
387  }
388 
389  constexpr double visual_length_thre = 1e-3; // [m]
390  if(child_bvh_joint->offset.norm() < visual_length_thre)
391  {
392  continue;
393  }
394 
395  rbd::parsers::Material::Color color;
396  color.r = 1.0;
397  color.g = 0.0;
398  color.b = 0.0;
399  color.a = 1.0;
400 
401  rbd::parsers::Visual rbd_visual_cylinder;
402  rbd_visual_cylinder.name = bvh_joint->name;
403  rbd_visual_cylinder.origin.rotation() = rotationMatrixFromZAxis(child_bvh_joint->offset).transpose();
404  rbd_visual_cylinder.origin.translation() = 0.5 * child_bvh_joint->offset;
405  rbd_visual_cylinder.geometry.type = rbd::parsers::Geometry::CYLINDER;
406  rbd::parsers::Geometry::Cylinder cylinder;
407  if(config_.urdf_cylinder_radius_list.count(child_bvh_joint->name))
408  {
409  cylinder.radius =
410  std::min(config_.urdf_cylinder_radius_list.at(child_bvh_joint->name), 0.5 * child_bvh_joint->offset.norm());
411  }
412  else
413  {
414  cylinder.radius = config_.urdf_cylinder_radius_scale_factor * child_bvh_joint->offset.norm();
415  }
416  cylinder.length = child_bvh_joint->offset.norm() - 2 * cylinder.radius;
417  rbd_visual_cylinder.geometry.data = cylinder;
418  rbd_visual_cylinder.material.type = rbd::parsers::Material::Type::COLOR;
419  rbd_visual_cylinder.material.data = color;
420  rbd_visual_list.push_back(rbd_visual_cylinder);
421 
422  for(int sphere_idx = 0; sphere_idx < 2; sphere_idx++)
423  {
424  rbd::parsers::Visual rbd_visual_sphere;
425  rbd_visual_sphere.name = bvh_joint->name;
426  rbd_visual_sphere.origin.rotation().setIdentity();
427  if(sphere_idx == 0)
428  {
429  rbd_visual_sphere.origin.translation() = cylinder.radius * child_bvh_joint->offset.normalized();
430  }
431  else
432  {
433  rbd_visual_sphere.origin.translation() =
434  (child_bvh_joint->offset.norm() - cylinder.radius) * child_bvh_joint->offset.normalized();
435  }
436  rbd_visual_sphere.geometry.type = rbd::parsers::Geometry::SPHERE;
437  rbd::parsers::Geometry::Sphere sphere;
438  sphere.radius = cylinder.radius;
439  rbd_visual_sphere.geometry.data = sphere;
440  rbd_visual_sphere.material.type = rbd::parsers::Material::Type::COLOR;
441  rbd_visual_sphere.material.data = color;
442  rbd_visual_list.push_back(rbd_visual_sphere);
443  }
444  }
445 
446  rbd_parser_result.visual.emplace(bvh_joint->name, rbd_visual_list);
447  }
